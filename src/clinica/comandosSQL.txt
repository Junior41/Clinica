CREATE DATABASE clinica;
USE clinica;
CREATE TABLE medico(CRM INTEGER NOT NULL, nome VARCHAR(20) NOT NULL, especialidade VARCHAR(20) NOT NULL, PRIMARY KEY(CRM));
CREATE TABLE filho(idFilho INTEGER AUTO_INCREMENT, nome VARCHAR(20) NOT NULL, dataNascimento DATETIME NOT NULL, sexo ENUM('M','F') NOT NULL, peso FLOAT NOT NULL, altura FLOAT NOT NULL, PRIMARY KEY(idFilho));
CREATE TABLE mae(CPF VARCHAR(11) NOT NULL, nome VARCHAR(20) NOT NULL, dataNacimento DATE NOT NULL, endereco VARCHAR(30), telefone VARCHAR(14) NOT NULL, PRIMARY KEY(CPF));
CREATE TABLE parto(CRM INTEGER NOT NULL, idFilho INTEGER NOT NULL, CPFMae VARCHAR(14) NOT NULL, PRIMARY KEY(CRM, idFilho, CPFMae), FOREIGN KEY(CRM) REFERENCES medico(CRM), FOREIGN KEY(idFilho) REFERENCES filho(idFilho), FOREIGN KEY(CPFMae) REFERENCES mae(CPF));
